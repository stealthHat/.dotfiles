1 - Kitty Terminal:
  $ mkdir -p $HOME/.config/kitty
  $ ln -sfv $HOME/git/.dotfiles/.config/kitty/kitty.conf $HOME/.config/kitty
  install kitty via package manager

2 - zsh
  $ mkdir -p $HOME/.config/zsh
  $ ln -sfv $HOME/git/.dotfiles/.config/zsh/.zshrc $HOME/.config/zsh
  $ ln -sfv $HOME/git/.dotfiles/.config/zsh/.zsh_aliases $HOME/.config/zsh
  $ export ZSH=$HOME/.config/zsh/.oh-my-zsh                                                                                 # Variavel de ambiente para que o oh-my-zsh seja installado em .config/zsh
  $ export ZPLUG_HOME=$HOME/.config/zsh/.zplug                                                                              # Variavel de ambiente para que o zplug seja installado em .config/zsh
  ? Install zsh via package manager
  $ sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"                         # Install oh-my-zsh
  $ curl -sL --proto-redir -all,https https://raw.githubusercontent.com/zplug/installer/master/installer.zsh | zsh          # Install zplug
  $ zplug install && zplug load 
  $ chsh -s $(which zsh)                                                                                                    # Seta o zsh como shell padrao caso o oh-my-zsh n√£o conseguir
 
3 - Nvim:
  $ mkdir -p $HOME/.config/nvim
  $ ln -sfv $HOME/git/.dotfiles/.config/nvim/init.vim $HOME/.config/nvim
  ? sudo <package-manager> install nvim
  $ curl -fLo ~/.local/share/nvim/site/autoload/plug.vim --create-dirs \                                                    # Install nvim plugin manager
       https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim
  ? Inside vim :PlugInstall

4 - Tmux
  $ ln -sfv $HOME/git/.dotfiles/.tmux.conf $HOME/
  ? Install tmux via package manager 
  $ git clone https://github.com/tmux-plugins/tpm ~/.config/tmux/plugins/tpm                                                # Install tmux plugin manager 
  ? Inside tmux: leader + I
