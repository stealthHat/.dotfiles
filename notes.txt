1 - Kitty Terminal:
  mkdir -p $HOME/.config/kitty
  ln -sfv $HOME/git/.dotfiles/.config/kitty/kitty.conf $HOME/.config/kitty
  install kitty via package manager

2 - zsh
  mkdir -p $HOME/.config/zsh
  ln -sfv $HOME/git/.dotfiles/.config/zsh/.zshrc $HOME/.config/zsh
  ln -sfv $HOME/git/.dotfiles/.config/zsh/.zsh_aliases $HOME/.config/zsh
  oh-my-zsh: export ZSH=~/.config/zsh/.oh-my-zsh 
  Zplug: export ZPLUG_HOME=$HOME/.config/zsh/.zplug
  Install zsh via package manager
  Install oh-my-zsh: sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"
  $ zplug install && zplug load 
  $ chsh -s $(which zsh) (seta o zsh como shell padrao)

3 - Nvim:
  mkdir -p $HOME/.config/nvim
  ln -sfv $HOME/git/.dotfiles/.config/nvim/init.vim $HOME/.config/nvim
  sudo <package-manager> install nvim
  Install vim-plug: curl -fLo ~/.local/share/nvim/site/autoload/plug.vim --create-dirs \
    https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim
  inside vim :PlugInstall

4 - Tmux
  ln -sfv $HOME/git/.dotfiles/.tmux.conf $HOME/
  Install tmux via package manager 
  Install tmux-plugin: git clone https://github.com/tmux-plugins/tpm ~/.config/tmux/plugins/tpm
  Inside tmux: leader + I 

5 - i3
  Install i3 via package manager
  $ mkdir -p $HOME/.config/i3
  $ ln -sfv $HOME/git/.dotfileS/.config/i3/config $HOME/.config/i3/
  reboot

6 - dmenu
  $ cd $HOME/git && git clone https://git.suckless.org/dmenu
  $ ln -sfv $HOME/git/dmenu/file $HOME/git/dmenu
