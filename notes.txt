  mkdir -p $HOME/.config/zsh
  ln -sfv $HOME/git/.dotfiles/.config/zsh/.zshrc $HOME/.config/zsh
  ln -sfv $HOME/git/.dotfiles/.config/zsh/.zsh_aliases $HOME/.config/zsh

  $zplug install
