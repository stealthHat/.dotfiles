Set necessary folders:
mkdir -p $HOME/.config/zsh
mkdir -p $HOME/.config/nvim
mkdir -p $HOME/.config/kitty

Set enviroments variables:
oh-my-zsh: export ZSH=~/.config/zsh/.oh-my-zsh 
zplug: export ZPLUG_HOME=$HOME/.config/zsh/.zplug

Set symblinks:
ln -s $HOME/git/.dotfiles/.zshrc $HOME/.config/zsh
ln -s $HOME/git/.dotfiles/.zsh_aliases $HOME/.config/zsh
ln -s $HOME/git/.dotfiles/init.vim $HOME/.config/nvim
ln -s $HOME/git/.dotfiles/kitty.conf $HOME/.config/kitty
ln -s $HOME/git/.dotfiles/.tmux.conf $HOME

1 - install kitty
2 - install zsh
3 - chsh -s $(which zsh) (seta o zsh como shell padrao)
4 - install oh-my-zsh: sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"
5 - install zplug: curl -sL --proto-redir -all,https https://raw.githubusercontent.com/zplug/installer/master/installer.zsh | zsh
6 - source $HOME/.config/zsh/.zshrc
7 - zplug install && zplug load
8 - install nvim: sudo pacman -Sy neovim
9 - install nvim plugin manager: curl -fLo ~/.local/share/nvim/site/autoload/plug.vim --create-dirs \
    https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim
10 - inside nvim run :PluginInstall
11 - install fira-code via package manager
12 - install tmux with the pakage manager
13 - install tmux plugin manager: git clone https://github.com/tmux-plugins/tpm ~/.config/tmux/plugins/tpm
