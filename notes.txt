Set necessary folders:
mkdir -p $HOME/.config/zsh
mkdir -p $HOME/.config/nvim

Set enviroments variables:
oh-my-zsh: export ZSH=~/.config/zsh/.oh-my-zsh 
zplug: export ZPLUG_HOME=$HOME/.config/zsh/.zplug

Set symblinks:
ln -s $HOME/git/.dotfiles/.zshrc $HOME/.config/zsh
ln -s $HOME/git/.dotfiles/.zsh_aliases $HOME/.config/zsh
ln -s $HOME/git/.dotfiles/init.vim $HOME/.config/nvim
ln -s $HOME/git/.dotfiles/.zshenv $HOME

1 - install zsh
2 - chsh -s $(which zsh) (seta o zsh como shell padrao)
3 - install oh-my-zsh: sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"
4 - install zplug: curl -sL --proto-redir -all,https https://raw.githubusercontent.com/zplug/installer/master/installer.zsh | zsh
5 - source $HOME/.config/zsh/.zshrc
6 - zplug install && zplug load
7 - install nvim: sudo pacman -Sy neovim
8 - install nvim plugin manager: curl -fLo ~/.local/share/nvim/site/autoload/plug.vim --create-dirs \
    https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim
9 - inside nvim run :PluginInstall
10 - install fira-code via package manager
